/**
 * 创建生成器错误
 * @param {string} code 错误代码
 * @param {string} [message] 错误消息
 * @param {Error} [cause] 原始错误
 * @returns {GeneratorError} 生成器错误
 */
function createGeneratorError(code, message = '', cause = null) {
  const errorMessage = message || ERROR_MESSAGES[code] || 'Unknown error'
  const track = `generator_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`
  const error = new GeneratorError(code, errorMessage, track)
  if (cause) {
    error.cause = cause
    error.stack = `${error.stack}\nCaused by: ${cause.stack}`
  }
  return error
}
 
class CodeGenerator {
  constructor(options = {}) {
    this.logger = new Logger({
      module: 'Generator',
      debug_enabled: process.env.NODE_ENV === 'development',
    })
    this.options = {
      templateDir: path.join(__dirname, '../templates/crud/base'),
      tempDir: path.join(__dirname, '../temp'),
      ...options,
    }
    this.templateEngine = new TemplateEngine()
    this.baseDir = path.join(__dirname, '../templates/crud/base')
  }
 
  /**
   * 初始化生成器
   */
  async init() {
    try {
      this.logger.debug('初始化生成器', { options: this.options })
 
      // 加载基础模板
      const layoutsPath = path.join(this.baseDir, 'common/layouts.edge')
      const layoutContent = await fsPromises.readFile(layoutsPath, 'utf8')
      this.templateEngine.loadBaseTemplate('common/layouts', layoutContent)
      this.logger.info('基础模板加载完成')
 
      // 确保临时目录存在
      if (!fs.existsSync(this.options.tempDir)) {
        this.logger.debug('创建临时目录', { dir: this.options.tempDir })
        fs.mkdirSync(this.options.tempDir, { recursive: true })
      }
 
      // 清理旧的临时文件
      await this.cleanTempDir()
 
      this.logger.info('生成器初始化完成')
    } catch (error) {
      this.logger.error('初始化生成器失败', { error })
      if (error instanceof GeneratorError) {
        throw error
      }
      throw createGeneratorError(ERROR_CODES.SYS_ERROR, '初始化生成器失败', error)
    }
  }
 
  /**
   * 生成代码
   */
  async generate(moduleName, sqlInfo, targetDir) {
    this.logger.debug('开始生成代码', { moduleName, sqlInfo, targetDir })
 
    try {
      // 生成前端文件
      await this.generateFrontend(moduleName, sqlInfo, targetDir)
 
      // 生成后端文件
      await this.generateBackend(moduleName, sqlInfo, targetDir)
 
      // 生成配置文件
      await this.generateConfig(moduleName, sqlInfo, targetDir)
 
      this.logger.info('代码生成完成')
    } catch (error) {
      this.logger.error('生成代码失败', { error, params: { moduleName, sqlInfo, targetDir } })
      throw error
    }
  }
 
  /**
   * 生成前端文件
   * @private
   */
  async generateFrontend(moduleName, sqlInfo, targetDir) {
    try {
      const frontendDir = path.join(targetDir, 'frontend')
      this.logger.debug('创建前端目录', { dir: frontendDir })
      await fsPromises.mkdir(frontendDir, { recursive: true })
 
      // 准备模板数据
      const data = await this._prepareTemplateData(moduleName, sqlInfo)
      this.logger.debug('准备模板数据', { data })
 
      // 生成前端文件
      await this._generateFile('frontend/list', path.join(frontendDir, 'list.edge'), data)
      await this._generateFile('frontend/create', path.join(frontendDir, 'create.edge'), data)
      await this._generateFile('frontend/edit', path.join(frontendDir, 'edit.edge'), data)
      await this._generateFile('frontend/view', path.join(frontendDir, 'view.edge'), data)
 
      this.logger.info('前端文件生成完成')
    } catch (error) {
      this.logger.error('生成前端文件失败', { error, workDir: targetDir })
      throw error
    }
  }

      await this._generateFile('frontend/edit', data, path.join(frontendDir, 'edit.edge'))

  /**
   * 生成单个文件
   * @private
   */
  async _generateFile(templateName, targetPath, data) {
    this.logger.debug('开始生成文件', { template: templateName, target: targetPath })
 
    // 获取模板文件路径
    const templatePath = path.join(this.baseDir, 'templates/crud/base', `${templateName}.tpl`)
 
    // 检查模板文件是否存在
    if (!fs.existsSync(templatePath)) {
      this.logger.error('模板文件不存在', { templatePath })
      throw new GeneratorError(ERROR_CODES.TEMPLATE_NOT_FOUND, `模板文件不存在: ${templatePath}`, 'generator_generateFile_template_not_found')
    }
 
    try {
      // 读取模板内容
      const template = await fsPromises.readFile(templatePath, 'utf8')
      this.logger.debug('模板内容读取成功', { template: templateName, length: template.length })
 
      // 渲染模板
      const content = this.templateEngine.render(template, data)
      this.logger.debug('模板渲染成功', { template: templateName, length: content.length })
 
      // 确保目标目录存在
      await fsPromises.mkdir(path.dirname(targetPath), { recursive: true })
 
      // 写入文件
      await fsPromises.writeFile(targetPath, content)
      this.logger.info('文件生成成功', { target: targetPath })
    } catch (error) {
      this.logger.error('生成文件失败', {
        error,
        template: templateName,
        target: targetPath,
        errorName: error.name,
        errorCode: error.code,
        errorMessage: error.message,
        errorStack: error.stack,
      })
      throw error
    }
  }
 
  /**
   * 准备模板数据
   * @private
   */
  async _prepareTemplateData(moduleName, sqlInfo) {
    try {
      this.logger.debug('开始准备模板数据', { moduleName, sqlInfo })
 
      // 验证字段
      if (!sqlInfo || !sqlInfo.fields || !Array.isArray(sqlInfo.fields)) {
        this.logger.error('字段信息无效', { sqlInfo })
        throw new GeneratorError(ERROR_CODES.SQL_PARSE_ERROR, '字段信息无效', 'generator_prepareTemplateData_missing_fields')
      }
      if (sqlInfo.fields.length === 0) {
        this.logger.error('字段列表为空', { fields: sqlInfo.fields })
        throw new GeneratorError(ERROR_CODES.SQL_PARSE_ERROR, '未找到任何字段', 'generator_prepareTemplateData_empty_fields')
      }
 
      // 准备数据
      const data = {
        module_name: moduleName,
        module_id_field: `${moduleName}_id`,
        table_name: sqlInfo.table,
        fields: sqlInfo.fields.map((field) => {
          const mappedField = {
            name: field.name,
            label: field.comment || field.name,
            type: this._mapFieldType(field.type),
            listable: field.listable !== false,
            filterable: field.filterable !== false,
            required: !field.nullable,
            min_width: field.min_width || '200px',
            sort_direction: field.sort_direction || 'asc',
            filter_type: field.filter_type || 'text',
          }
          this.logger.debug('字段映射结果', {
            original: field,
            mapped: mappedField,
          })
          return mappedField
        }),
        has_create_permission: true,
        has_edit_permission: true,
        has_delete_permission: true,
        has_batch_delete_permission: true,
        options: sqlInfo.options || {
          status: [
            { value: 1, label: "trans('enable')" },
            { value: 0, label: "trans('disable')" },
          ],
        },
      }
 
      this.logger.debug('模板数据准备完成', { data })
      return data
    } catch (error) {
      this.logger.error('准备模板数据失败', {
        error,
        moduleName,
        sqlInfo,
        errorName: error.name,
        errorCode: error.code,
        errorMessage: error.message,
        errorStack: error.stack,
      })
      if (error instanceof GeneratorError) {
        throw error
      }
      throw new GeneratorError(ERROR_CODES.SQL_PARSE_ERROR, `准备模板数据失败: ${error.message}`, 'generator_prepareTemplateData_error')
    }
  }
 
  /**
   * 映射字段类型
   * @private
   */
  _mapFieldType(sqlType) {
    this.logger.debug('开始映射字段类型', { sqlType })
 
    if (!sqlType) {
      this.logger.warn('SQL类型为空，使用默认类型text', { sqlType })
      return 'text'
    }
 
    const typeMap = {
      int: 'number',
      tinyint: 'number',
      smallint: 'number',
      mediumint: 'number',
      bigint: 'number',
      float: 'number',
      double: 'number',
      decimal: 'number',
      varchar: 'text',
      char: 'text',
      text: 'text',
      mediumtext: 'text',
      longtext: 'text',
      datetime: 'datetime',
      timestamp: 'datetime',
      date: 'date',
      time: 'time',
      year: 'year',
      boolean: 'boolean',
      json: 'json',
      blob: 'blob',
      mediumblob: 'blob',
      longblob: 'blob',
    }
 
    const mappedType = typeMap[sqlType.toLowerCase()] || 'text'
    this.logger.debug('字段类型映射结果', {
      sqlType,
      mappedType,
      isDefaultType: !typeMap[sqlType.toLowerCase()],
    })
    return mappedType
  }

  /**
   * 清理临时目录
   * @private
   */
  async cleanTempDir(dir = this.options.tempDir) {
    try {
      if (fs.existsSync(dir)) {
        await fsPromises.rm(dir, { recursive: true, force: true })
      }
      if (dir === this.options.tempDir) {
        fs.mkdirSync(dir, { recursive: true })
      }
    } catch (error) {
      throw new GeneratorError('清理临时目录失败', { cause: error })
    }
  }
 
  /**
   * 移动文件
   * @private
   */
  async moveFiles(sourceDir, targetDir) {
    try {
      // 确保目标目录存在
      fs.mkdirSync(targetDir, { recursive: true })
 
      // 递归复制文件
      const copyRecursive = async (src, dest) => {
        const stats = await fsPromises.stat(src)
        if (stats.isDirectory()) {
          // 创建目标目录
          fs.mkdirSync(dest, { recursive: true })
          // 读取源目录内容
          const entries = await fsPromises.readdir(src, { withFileTypes: true })
          // 递归复制每个条目
          for (const entry of entries) {
            await copyRecursive(path.join(src, entry.name), path.join(dest, entry.name))
          }
        } else {
          // 复制文件
          await fsPromises.copyFile(src, dest)
        }
      }
 
      await copyRecursive(sourceDir, targetDir)
    } catch (error) {
      throw new GeneratorError('移动文件失败', { cause: error })
    }
  }
 
  /**
   * 列出生成的文件
   * @private
   */
  async listGeneratedFiles(dir) {
    try {
      const files = []
      const listRecursive = async (currentDir) => {
        const entries = await fsPromises.readdir(currentDir, { withFileTypes: true })
        for (const entry of entries) {
          const fullPath = path.join(currentDir, entry.name)
          if (entry.isDirectory()) {
            await listRecursive(fullPath)
          } else {
            files.push(path.relative(dir, fullPath))
          }
        }
      }
      await listRecursive(dir)
      return files
    } catch (error) {
      throw new GeneratorError('列出生成的文件失败', { cause: error })
    }
  }
